177:   def parse_time(input, minutes = false)
178:     total = 0
179:     unless input.nil?
180:       miss = false
181:       reg = Regexp.new("(#{_('[wdhm]')})")
182:       input.downcase.gsub(reg,'\1 ').split(' ').each do |e|
183:         part = /(\d+)(\w+)/.match(e)
184:         if part && part.size == 3
185:           case  part[2]
186:           when _('w') then total += e.to_i * current_user.workday_duration * current_user.days_per_week
187:           when _('d') then total += e.to_i * current_user.workday_duration
188:           when _('h') then total += e.to_i * 60
189:           when _('m') then total += e.to_i
190:           else 
191:             miss = true
192:           end
193:         end
194:       end
195: 
196:       # Fallback to default english parsing
197:       if miss
198:         eng_total = 0

200:         input.downcase.gsub(reg,'\1 ').split(' ').each do |e|
201:           part = /(\d+)(\w+)/.match(e)
202:           if part && part.size == 3
203:             case  part[2]
204:             when 'w' then eng_total += e.to_i * current_user.workday_duration * current_user.days_per_week
205:             when 'd' then eng_total += e.to_i * current_user.workday_duration

207:             when 'm' then eng_total += e.to_i
208:             end
209:           end
210:         end
211:

216:       end
