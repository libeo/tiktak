def get_watchers 

  @users = Project.find(params[:project_id]).users.find(:all, :order => 'name' )

  @options = @users.collect{ |u| (' {"text":"' + u.name.gsub(/"/,'\"') + '", "value":"' + u.id.to_s + '"}') }.join( ',' )
  res = '{"options":['
  res << "#{@options}" unless @options.nil? || @options.empty?
  res << ']}'

  render :text => "#{res}"

end
